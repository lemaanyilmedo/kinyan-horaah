  const wrongSound = new Audio('/wp-content/uploads/2025/07/wrong.mp3');
  const levelUpSound = new Audio('/wp-content/uploads/2025/07/levelUp.mp3');
  const clappingSound = new Audio('/wp-content/uploads/2025/07/clapping.mp3');
  const backgroundMusic = new Audio('/wp-content/uploads/2025/07/loopMusic.mp3');

  const sounds = [tickSound, correctSound, wrongSound, levelUpSound, clappingSound, backgroundMusic];
  let isMuted = false;
  let userInteracted = false;

  let gameStartTime = null;

  backgroundMusic.volume = 0.3;
  backgroundMusic.loop = true;

  muteBtn.addEventListener('click', () => {
    isMuted = !isMuted;
    sounds.forEach(s => s.muted = isMuted);
    muteBtn.style.opacity = isMuted ? '0.5' : '1';
  });

  let questionsByGrade = {};
  let currentGradeIndex = 0;
  let currentQuestions = [];
  let correctInGrade = 0;
  let askedInGrade = 0;
  let score = 0;
  let timerId = null;

  const maxCorrectPerGrade = 6;
  const maxAskedPerGrade = 10;
  const questionTime = 15;

  fetch(gameUrl).then(r => r.json()).then(initGame);

  function initGame(data) {
    rankEl.textContent = '0';
    data.forEach(q => {
      (questionsByGrade[q.grade] ||= []).push(q);
    });
    gameStartTime = Date.now();
    if (!isMuted) backgroundMusic.play();
    startGrade();
  }

  function showLevelUpMessage(done) {
    const lvlBox = document.querySelector('.levelupContainer');
    if (!lvlBox) { done(); return; }
    const qLines = document.querySelector('.questionLines');
    const ansBox = document.querySelector('.answers');
    qLines.style.display = 'none';
    ansBox.style.display = 'none';
    lvlBox.innerHTML = '';
    const name = localStorage.getItem('gamePlayerName') || '';
    lvlBox.innerHTML = `
      <div class="levelup-message">${name ? name + ',' : ''} כל הכבוד!</div>
      <div class="levelup-subline">עלית כיתה!</div>
      <img class="levelup-image" src="/wp-content/uploads/2025/07/wellDone.svg" alt="well done">
    `;
    if (!isMuted) levelUpSound.play();
    setTimeout(() => {
      lvlBox.innerHTML = '';
      qLines.style.display = '';
      ansBox.style.display = '';
      done();
    }, 3000);
  }

  function startGrade() {
    const grade = gradeOrder[currentGradeIndex];
    correctInGrade = 0;
    askedInGrade = 0;
    currentQuestions = [...(questionsByGrade[grade] || [])];
    for (let i = 1; i <= maxCorrectPerGrade; i++) {
      const el = document.querySelector(`.number${i}`);
      if (el) el.innerHTML = `<img src="/wp-content/uploads/2025/07/number${i}.svg" alt="${i}">`;
    }
    updateGradeDisplay(grade);
    nextQuestion();
  }

  function updateGradeDisplay(grade) {
    gradeEl.textContent = `כיתה ${grade}`;
  }

  function nextQuestion() {
    clearInterval(timerId);
    if (correctInGrade >= maxCorrectPerGrade) {
      currentGradeIndex++;
      const available = gradeOrder.filter(g => questionsByGrade[g] && questionsByGrade[g].length);
      if (currentGradeIndex >= available.length) {
        localStorage.setItem('gamePlayerScore', score.toString());
        const durationSec = Math.round((Date.now() - gameStartTime) / 1000);
        localStorage.setItem('gameDuration', durationSec.toString());
        if (!isMuted) clappingSound.play();
        setTimeout(() => window.location.href = '/king/', 2000);
        return;
      }
      showLevelUpMessage(startGrade);
      return;
    }
    if (askedInGrade >= maxAskedPerGrade || currentQuestions.length === 0) {
      gameOver();
      return;
    }
    const idx = Math.floor(Math.random() * currentQuestions.length);
    const q = currentQuestions.splice(idx, 1)[0];
    askedInGrade++;
    renderQuestion(q);
    startTimer(() => {
      if (!isMuted) tickSound.play();
      nextQuestion();
    });
  }

  function renderQuestion(q) {
    questionEl.textContent = q.question;
    answersEl.innerHTML = '';
    q.options.forEach((opt, i) => {
      const btn = document.createElement('button');
      btn.className = 'answerBtn';
      btn.textContent = opt;
      btn.dataset.correct = q.correctIndex === i ? '1' : '0';
      btn.addEventListener('click', answerHandler);
      answersEl.appendChild(btn);
    });
  }

  function answerHandler(e) {
    if (!userInteracted) {
      userInteracted = true;
      try { tickSound.play().then(() => { tickSound.pause(); tickSound.currentTime = 0; }); } catch (err) {}
    }
    clearInterval(timerId);
    const isCorrect = e.currentTarget.dataset.correct === '1';
    document.querySelectorAll('.answerBtn').forEach(b => b.disabled = true);
    if (isCorrect) {
      if (!isMuted) correctSound.play();
      score += 10;
      correctInGrade++;
      rankEl.textContent = score.toString();
      e.currentTarget.style.backgroundColor = '#D8E9D2';
      e.currentTarget.style.border = '2px solid #41AD49';
      e.currentTarget.style.color = '#41AD49';
      const numEl = document.querySelector(`.number${correctInGrade}`);
      if (numEl) numEl.innerHTML = '<img src="/wp-content/uploads/2025/07/doneIcon.svg" alt="done">';
    } else {
      if (!isMuted) wrongSound.play();
      e.currentTarget.style.backgroundColor = '#F3C1C1';
      e.currentTarget.style.border = '2px solid #C6191C';
      e.currentTarget.style.color = '#C6191C';
    }
    setTimeout(nextQuestion, 1000);
  }

  function startTimer(cb) {
    let t = questionTime;
    const circle = document.querySelector('.progress');
    const timeText = document.querySelector('.timeText');
    const R = 45, C = 2 * Math.PI * R;
    circle.style.strokeDasharray = C;
    circle.style.strokeDashoffset = 0;
    timeText.textContent = t;
    timerId = setInterval(() => {
      t--;
      timeText.textContent = t;
      circle.style.strokeDashoffset = C - (t / questionTime) * C;
      if (t <= 0) { clearInterval(timerId); cb(); }
    }, 1000);
  }

  function gameOver() {
    clearInterval(timerId);
    localStorage.setItem('gamePlayerScore', score.toString());
    const name = localStorage.getItem('gamePlayerName') || '';
    localStorage.setItem('gamePlayerName', name);
    const durationSec = Math.round((Date.now() - gameStartTime) / 1000);
    localStorage.setItem('gameDuration', durationSec.toString());
    if (!isMuted) clappingSound.play();
    setTimeout(() => window.location.href = '/gameover/', 2000);
  }
});

document.addEventListener('DOMContentLoaded', function () {
		
	
    function ensurePlayerId() {

    }

    const playerName = localStorage.getItem('gamePlayerName') || '';

    function getScoreKey(){

    }

    function getDurationKey(){
        return 'gameTotalDuration_' + playerId;
    }

    function getScore(){
        return parseInt(localStorage.getItem(getScoreKey()) || '0', 10);
    }

    function setScore(v){
        localStorage.setItem(getScoreKey(), v.toString());
        localStorage.setItem('gamePlayerScore', v.toString());
    }

    function getDuration(){
        return parseInt(localStorage.getItem(getDurationKey()) || '0', 10);
    }

    function setDuration(v){
        localStorage.setItem(getDurationKey(), v.toString());
        localStorage.setItem('gameDuration', v.toString());
    }

    window.osherGameIdentity = {

document.addEventListener('DOMContentLoaded', function() {

    if (window.osherGameLoaded) return;
    window.osherGameLoaded = true;

    var origPlay = HTMLMediaElement.prototype.play;
    HTMLMediaElement.prototype.play = function() {
        try {
            if (this.src && this.src.indexOf('/wp-content/uploads/2025/07/loopMusic.mp3') !== -1) {
                if (window.__osherLoopAudio && window.__osherLoopAudio !== this) {
                    window.__osherLoopAudio.pause();
                }
                window.__osherLoopAudio = this;
            }
        } catch (e) {}
        return origPlay.apply(this, arguments);
    };

    const gameUrl = '/wp-content/uploads/game/game.json';
    const gameContainer = document.querySelector('.gameContainer');
    if (!gameContainer) return;

    const questionTitle =
        gameContainer.querySelector('.questionLines .elementor-heading-title') ||
        gameContainer.querySelector('.question .elementor-heading-title');

    const answersEl = gameContainer.querySelector('.answers');
    const rankContainer = gameContainer.querySelector('.rank');
    const playerNameEl = gameContainer.querySelector('.playerNameHere');
    const timeLine = gameContainer.querySelector('.timeLine');
    const muteBtn = gameContainer.querySelector('.mute') || document.querySelector('.mute');

    let progressFill = null;

    if (timeLine) {

            '<div class="timeline-wrapper">' +

        currentQuestionEl = timeLine.querySelector('.timeline-current');
    }

    const tickSound       = new Audio('https://actions.google.com/sounds/v1/alarms/beep_short.ogg');
    const correctSound    = new Audio('/wp-content/uploads/2025/07/correct.mp3');

    const backgroundMusic = new Audio('/wp-content/uploads/2025/07/loopMusic.mp3');

    const sounds = [
        tickSound, correctSound, wrongSound,
        levelUpSound, clappingSound, backgroundMusic
    ];

    let isMuted = false;
    let userInteracted = false;
    let gameStartTime = null;
    let questionsPool = [];
    let questionsSequence = [];

    let score = osherGameIdentity.getScore();
    let timerId = null;

    const totalQuestionsPerGame = 24;
    const questionTime = 24;

    backgroundMusic.volume = 0.3;
    backgroundMusic.loop = true;

    if (muteBtn) {
        muteBtn.addEventListener('click', function() {
            isMuted = !isMuted;
            sounds.forEach(s => s.muted = isMuted);
            muteBtn.style.opacity = isMuted ? '0.5' : '1';
            if (isMuted) backgroundMusic.pause();
            else backgroundMusic.play().catch(()=>{});
        });
    }

    if (nextBtn) {
        nextBtn.addEventListener('click', function() {
            clearInterval(timerId);
            answeredCount++;
            pointer++;
            if (!isMuted) wrongSound.play();
            updateTimeline();
            if (answeredCount >= totalQuestionsPerGame) {
                setTimeout(endGame, 800);
                return;
            }
            setTimeout(showNewQuestion, 800);
        });

    const storedName = localStorage.getItem('gamePlayerName') || '';

    function shuffleArray(a) {
        for (let i = a.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            const t = a[i];
            a[i] = a[j];
            a[j] = t;
        }
        return a;
    }

    function getUsedQuestions() {
        try {
            return JSON.parse(localStorage.getItem('gameUsedQuestions') || '[]');
        } catch (e) { return [] }
    }

    function saveUsedQuestions(a) {
        localStorage.setItem('gameUsedQuestions', JSON.stringify(a));
    }

    fetch(gameUrl)
        .then(r => r.json())
        .then(initGame);

    function initGame(data) {
        if (!Array.isArray(data) || !data.length) return;

        questionsPool = data.slice();
        let used = getUsedQuestions();
        let unused = questionsPool.filter(q => !used.includes(q.id));

        if (unused.length < totalQuestionsPerGame) {
            used = [];
            saveUsedQuestions([]);
            unused = questionsPool.slice();
        }

        unused = shuffleArray(unused);
        questionsSequence = unused.slice();

        const ids = questionsSequence.map(q => q.id);
        saveUsedQuestions(used.concat(ids));

				pointer = 0;
				answeredCount = 0;
				gameStartTime = Date.now();

				osherGameIdentity.setDuration(0);

				score = osherGameIdentity.getScore();
				updateScoreDisplay(score);

        if (progressFill) progressFill.style.width = '0%';
        if (currentQuestionEl) {
            currentQuestionEl.textContent = '1';
            currentQuestionEl.style.right = '0%';
        }

        if (!isMuted) backgroundMusic.play().catch(()=>{});

        showNewQuestion();
    }

    function showNewQuestion() {
        clearInterval(timerId);

        if (answeredCount >= totalQuestionsPerGame) {
            endGame();
            return;
        }

        if (pointer >= questionsSequence.length) {
            pointer = 0;
        }

        const q = questionsSequence[pointer];
        renderQuestion(q);

        startTimer(() => {
            pointer++;
            showNewQuestion();
        });
    }

    function renderQuestion(q) {
        if (questionTitle) questionTitle.textContent = q.question;
        if (!answersEl) return;

        answersEl.innerHTML = '';

        q.options.forEach(function(opt, i) {
            const b = document.createElement('button');
            b.className = 'answerBtn';
            b.textContent = opt;
            b.dataset.correct = q.correctIndex === i ? '1' : '0';
            b.addEventListener('click', answerHandler);
            answersEl.appendChild(b);
        });
    }

    function answerHandler(e) {
        if (!userInteracted) {
            userInteracted = true;
            try {
                tickSound.play().then(function() {
                    tickSound.pause();
                    tickSound.currentTime = 0;
                });
            } catch (e) {}
        }

        clearInterval(timerId);

        const correct = e.currentTarget.dataset.correct === '1';

        gameContainer.querySelectorAll('.answerBtn').forEach(b => b.disabled = true);

        if (correct) {
            if (!isMuted) correctSound.play();
            score += 10;
            osherGameIdentity.setScore(score);

            e.currentTarget.style.backgroundColor = '#D8E9D2';
            e.currentTarget.style.border = '2px solid #41AD49';
            e.currentTarget.style.color = '#41AD49';

            updateScoreDisplay(score);

        } else {
            if (!isMuted) wrongSound.play();
            e.currentTarget.style.backgroundColor = '#F3C1C1';
            e.currentTarget.style.border = '2px solid #C6191C';
            e.currentTarget.style.color = '#C6191C';
        }

        answeredCount++;
        pointer++;
        updateTimeline();

        if (answeredCount >= totalQuestionsPerGame) {
            setTimeout(endGame, 800);
            return;
        }

        setTimeout(showNewQuestion, 800);
    }

    function updateTimeline() {
        if (!progressFill || !currentQuestionEl) return;
        if (answeredCount >= totalQuestionsPerGame) return;

        const pct = (answeredCount / totalQuestionsPerGame) * 100;
        const num = answeredCount === 0 ? 1 : answeredCount + 1;

        progressFill.style.width = pct + '%';
        currentQuestionEl.textContent = num;

        if (answeredCount === 0) {
            currentQuestionEl.style.right = '0%';
        } else {
            currentQuestionEl.style.right = pct + '%';
        }
    }

    function updateScoreDisplay(v) {
        if (!rankContainer) return;

        const padded = v.toString().padStart(4, '0');
        const digits = padded.split('');

        const idx = digits.findIndex(x => x !== '0');
        let html = '';

        for (let i = 0; i < digits.length; i++) {
            const ch = digits[i];
            const zero = idx === -1 || i < idx;
            const c = zero ? '#ff2a3269' : '#FF2A32';
            const o = zero ? '0.5' : '1';

            html +=
                '<span class="rank-digit" style="color:' + c +
                ';opacity:' + o +
                ';display:inline-block;min-width:0.6em;text-align:center;">' +
                ch +
                '</span>';
        }

        rankContainer.innerHTML = html;
    }

                cb();
            }
        }, 1000);
